518  #define KSI_IMPLEMENT_OBJECT_PARSE(type, tag) \
519  int type##_parse(KSI_CTX *ctx, const unsigned char *raw, size_t len, type **t) { \
520  int res = KSI_UNKNOWN_ERROR; \
521  KSI_TLV *tlv = NULL; \
522  type *tmp = NULL; \
523  if (ctx == NULL || t == NULL) { \
524  res = KSI_INVALID_ARGUMENT; \
525  goto cleanup; \
526  } \
527  res = KSI_TLV_parseBlob2(ctx, (unsigned char *)raw, len, 0, &tlv); \
528  if (res != KSI_OK) goto cleanup; \
529  if (KSI_TLV_getTag(tlv) != (tag)) { \
530  res = KSI_INVALID_FORMAT; \
531  goto cleanup; \
532  } \
533  res = type##_new(ctx, &tmp); \
534  if (res != KSI_OK) goto cleanup; \
535  res = KSI_TlvTemplate_parse(ctx, raw, len, KSI_TLV_TEMPLATE(type), tmp); \
536  if (res != KSI_OK) goto cleanup; \
537  *t = tmp; \
538  tmp = NULL; \
539  res = KSI_OK; \
540  cleanup: \
541  KSI_TLV_free(tlv); \
542  type##_free(tmp); \
543  return res; \
544  } \
